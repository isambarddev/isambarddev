DECLARE @Version NVARCHAR(MAX)

DECLARE @Web NVARCHAR(MAX)

DECLARE @Session NVARCHAR(MAX)

DECLARE @master NVARCHAR(MAX)

DECLARE @core NVARCHAR(MAX)

DECLARE @integration NVARCHAR(MAX)

SET @Web = N’B:\sqlbackups\ARM-PRD-SQL-14\’ + @Version + N’\web.bak’

SET @Session = N’B:\sqlbackups\ARM-PRD-SQL-14\’ + @Version + N’\sessions.bak’
